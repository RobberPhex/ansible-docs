Examples
--------

 ::

    - name: BIGIP F5 | Create HTTP Monitor
      local_action:
        module:             bigip_monitor_http
        state:              present
        server:             "{{ f5server }}"
        user:               "{{ f5user }}"
        password:           "{{ f5password }}"
        name:               "{{ item.monitorname }}"
        send:               "{{ item.send }}"
        receive:            "{{ item.receive }}"
      with_items: f5monitors
    - name: BIGIP F5 | Remove HTTP Monitor
      local_action:
        module:             bigip_monitor_http
        state:              absent
        server:             "{{ f5server }}"
        user:               "{{ f5user }}"
        password:           "{{ f5password }}"
        name:               "{{ monitorname }}"
